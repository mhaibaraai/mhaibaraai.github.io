Mapbox GL JS v3 ​

Mapbox GL JS v3 支持 Mapbox 标准样式 和 Mapbox 标准卫星样式。

  • mapbox://styles/mapbox/standard
  • mapbox://styles/mapbox/standard-satellite

更改地图的样式配置属性 ​

参考 Mapbox 样式配置属性

vue
<script lang="ts" setup>
import type { _MapboxProps } from '@movk-repo/mapbox'
import { Mapbox } from '@movk-repo/mapbox'
import { type Map, NavigationControl } from 'mapbox-gl'

const accessToken = import.meta.env.VITE_MAPBOX_ACCESS_TOKEN

const options = ref({
  center: [2.2932, 48.86069],
  zoom: 15.1,
  pitch: 62,
  bearing: -20,
} as _MapboxProps['options'])

const lightPreset = computed(() => {
  const hour = new Date().getHours()
  if (hour >= 6 && hour < 8)
    return 'dawn'
  if (hour >= 8 && hour < 18)
    return 'day'
  if (hour >= 18 && hour < 20)
    return 'dusk'
  return 'night'
})

function load(map: Map) {
  map.addControl(new NavigationControl())
  map.setConfig('basemap', {
    lightPreset,
    showPointOfInterestLabels: false,
  })
}
</script>

<template>
  <Mapbox
    type="standard"
    :access-token="accessToken"
    :options="options"
    @load="load"
  />
</template>
